nginx.ingress.kubernetes.io/proxy-set-header: | 
  Host $host 
  X-Real-IP $remote_addr 
  X-Forwarded-For $proxy_add_x_forwarded_for X-Forwarded-Proto $scheme

  apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  annotations:
    nginx.ingress.kubernetes.io/proxy-set-header: |
      Host $host
      X-Real-IP $remote_addr
      X-Forwarded-For $proxy_add_x_forwarded_for
      X-Forwarded-Proto $scheme
spec:
  rules:
  - host: yourdomain.com
    http:
      paths:
      - pathType: Prefix
        backend:
          service:
            name: my-django-app-service
            port:
              number: 80
